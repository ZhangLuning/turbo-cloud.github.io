快速开始

阐述 Turbo CLoud 完善的微服务架构，在此之前，我们假设您已经：

  • 拥有 Java 开发环境以及相应 IDE
  • 熟悉 Spring Boot、Spring Cloud、MyBatis、MyBatis-Plus、Nacos、Vue、MySQL
  • 熟悉 Maven

# 框架结构图

  • Turbo Cloud项目结构说明如下：
序号 模块名称 说明
1 turbo-cloud-calc 计算模块，一般金融系统会使用到
2 turbo-cloud-core 框架核心包，工具类、切面、配置类、拦截器
3 turbo-cloud-cron 定时任务
4 turbo-cloud-es Elasticsearch搜索引擎
5 turbo-cloud-flow 工作流
6 turbo-cloud-generator 代码生成器
7 turbo-cloud-minio MinIO对象文件存储
7 turbo-cloud-openfeign 微服务间调用
8 turbo-cloud-order Demo案例订单服务
9 turbo-cloud-rule 规则引擎
10 turbo-cloud-starter Spring Cloud
11 turbo-cloud-ums 用户管理，用户权限

重要提示

注意 Spring Boot、Spring Cloud、Alibaba Cloud、Mybatis Plus要找到正确的对应版本。最新的Springboot:last版本(会持续更新)。

# 服务结构图

  • Turbo Cloud xx Service 项目结构说明如下：
序号 模块名称 说明
1 captcha 三方模块（图形验证码，此服务用到了）
2 config 配置模块
3 constants 常量模块
4 utils 工具类（业务处理）
5 web web层
6 web-client 微服务间调用层
7 web-entity 数据模型层（对应数据库表）
8 web-logic 多表，三方对接操作（控制事务）
9 web-mapper 数据存储对象操作层
9 web-model 数据模型层
11 web-service 业务逻辑层
11 web-webservice web控制层

重要提示

服务工程结构保持一直，方便日后维护和开发之间项目分享代码。

# 服务配置文件

  • 在 application.yml 微服务配置文件：
server:
  port: xxxxxx

spring:
  application:
    name: turbo-cloud-xxx
  profiles:
    active: dev

mybatis-plus:
  mapper-locations: classpath*:/mapper/**/*.xml
  #实体扫描，多个package用逗号或者分号分隔
  type-aliases-package: com.turbo.cloud.**.entity, com.turbo.cloud.**.mapper
  global-config:
    db-config:
      #主键类型  0:"数据库ID自增", 1:"用户输入ID",2:"全局唯一ID (数字类型唯一ID)", 3:"全局唯一ID UUID";
      id-type: ASSIGN_UUID
      #字段策略 0:"忽略判断",1:"非 NULL 判断"),2:"非空判断"
      insert-strategy: NOT_NULL
      update-strategy: NOT_NULL
      where-strategy: NOT_EMPTY
      #logic-delete-field: deletedTimeLong  # 全局逻辑删除的实体字段名(since 3.3.0,配置后可以忽略不配置步骤2)
      #logic-delete-value: unix_timestamp(now(6))*1000000 # 逻辑已删除值(默认为 1)
      #logic-not-delete-value: 0 # 逻辑未删除值(默认为 0)
  configuration:
    # 是否开启自动驼峰命名规则（camel case）映射
    map-underscore-to-camel-case: true
    cache-enabled: false
    jdbc-type-for-null: null
    call-setters-on-nulls: false
    local_cache_scope: STATEMENT

# 服务日志配置

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE configuration>

<configuration scan="true">
    <conversionRule conversionWord="trace" converterClass="com.turbo.cloud.core.trace.TraceConverter"/>
    <springProperty scope="context" name="springAppName" source="spring.application.name"/>
    <property name="CONSOLE_LOG_PATTERN"
              value="%clr(%d{${LOG_DATEFORMAT_PATTERN:-yyyy-MM-dd HH:mm:ss.SSS}}){faint} %clr(${LOG_LEVEL_PATTERN:-%5p}) %clr([%t]){faint} %trace %clr(%logger{39}[%line]){cyan} %clr(:){faint} %m%n${LOG_EXCEPTION_CONVERSION_WORD:-%wEx}"/>
    <include resource="org/springframework/boot/logging/logback/base.xml"/>
    <!-- The FILE and ASYNC appenders are here as examples for a production configuration -->
    <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>
                ${user.home}/data/logFiles/${springAppName}/%d{yyyy-MM-dd}/${springAppName}-logFile.%d{yyyy-MM-dd}.log
            </fileNamePattern>
            <maxHistory>90</maxHistory>
        </rollingPolicy>
        <encoder>
            <charset>utf-8</charset>
            <Pattern>%d{${LOG_DATEFORMAT_PATTERN:-yyyy-MM-dd HH:mm:ss.SSS}} ${LOG_LEVEL_PATTERN:-%5p} [%t] %trace
                %logger{39}[%line] : %m%n${LOG_EXCEPTION_CONVERSION_WORD:-%wEx}
            </Pattern>
        </encoder>
    </appender>

    <appender name="ERROR" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <filter class="ch.qos.logback.classic.filter.ThresholdFilter">
            <level>ERROR</level>
        </filter>
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>
                ${user.home}/data/logFiles/${springAppName}/%d{yyyy-MM-dd}/${springAppName}-logError.%d{yyyy-MM-dd}.log
            </fileNamePattern>
            <maxHistory>90</maxHistory>
        </rollingPolicy>
        <encoder>
            <charset>utf-8</charset>
            <pattern>%d{${LOG_DATEFORMAT_PATTERN:-yyyy-MM-dd HH:mm:ss.SSS}} ${LOG_LEVEL_PATTERN:-%5p} [%t] %trace
                %logger{39}[%line] : %m%n${LOG_EXCEPTION_CONVERSION_WORD:-%wEx}
            </pattern>
        </encoder>
    </appender>
    <appender name="WARN" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <filter class="ch.qos.logback.classic.filter.ThresholdFilter">
            <level>WARN</level>
        </filter>
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>
                ${user.home}/data/logFiles/${springAppName}/%d{yyyy-MM-dd}/${springAppName}-logWarn.%d{yyyy-MM-dd}.log
            </fileNamePattern>
            <maxHistory>90</maxHistory>
        </rollingPolicy>
        <encoder>
            <charset>utf-8</charset>
            <pattern>%d{${LOG_DATEFORMAT_PATTERN:-yyyy-MM-dd HH:mm:ss.SSS}} ${LOG_LEVEL_PATTERN:-%5p} [%t] %trace
                %logger{39}[%line] : %m%n${LOG_EXCEPTION_CONVERSION_WORD:-%wEx}
            </pattern>
        </encoder>
    </appender>

    <!-- 异步输出 -->
    <appender name="ASYNC" class="ch.qos.logback.classic.AsyncAppender">
        <!-- 不丢失日志.默认的,如果队列的80%已满,则会丢弃TRACT、DEBUG、INFO级别的日志 -->
        <discardingThreshold>0</discardingThreshold>
        <!-- 更改默认的队列的深度,该值会影响性能.默认值为256 -->
        <queueSize>512</queueSize>
        <!-- 添加附加的appender,最多只能添加一个 -->
        <appender-ref ref="FILE"/>
    </appender>

    <logger name="org.hibernate.validator" level="WARN"/>
    <logger name="org.springframework" level="WARN"/>
    <logger name="org.springframework.web" level="WARN"/>
    <logger name="org.springframework.security" level="WARN"/>
    <logger name="org.springframework.cache" level="WARN"/>
    <logger name="sun.rmi" level="WARN"/>


    <!-- https://logback.qos.ch/manual/configuration.html#shutdownHook and https://jira.qos.ch/browse/LOGBACK-1090 -->
    <shutdownHook class="ch.qos.logback.core.hook.DelayingShutdownHook"/>

    <contextListener class="ch.qos.logback.classic.jul.LevelChangePropagator">
        <resetJUL>true</resetJUL>
    </contextListener>

    <root level="#logback.loglevel#">
        <appender-ref ref="ASYNC"/>
        <appender-ref ref="CONSOLE"/>
        <appender-ref ref="ERROR"/>
        <appender-ref ref="WARN"/>
    </root>

</configuration>
